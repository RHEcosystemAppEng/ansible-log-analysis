**** Ansible Log Annotation Interface has been deployed! ****

1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "annotation-interface.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "Access the annotation interface at: http://$NODE_IP:$NODE_PORT"
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch its status by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "annotation-interface.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "annotation-interface.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo "Access the annotation interface at: http://$SERVICE_IP:{{ .Values.service.port }}"
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "annotation-interface.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Access the annotation interface at: http://127.0.0.1:7861"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 7861:$CONTAINER_PORT
{{- end }}

2. Monitor the deployment:
   kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "annotation-interface.name" . }},app.kubernetes.io/instance={{ .Release.Name }}"

3. View logs:
   kubectl logs --namespace {{ .Release.Namespace }} deployment/{{ include "annotation-interface.fullname" . }}

{{- if .Values.autoscaling.enabled }}
4. Horizontal Pod Autoscaler is enabled:
   kubectl get hpa --namespace {{ .Release.Namespace }} {{ include "annotation-interface.fullname" . }}
{{- end }}


For more information about using the annotation interface, please refer to the documentation.
